```js
/**
 * –†–∞–∑–±–∏–≤–∞–µ—Ç —á–∏—Å–ª–æ –Ω–∞ —á–∞—Å—Ç–∏ –∑–∞–¥–∞–Ω–Ω–æ–π –¥–ª–∏–Ω—ã
 * @param {number|string} n - –ß–∏—Å–ª–æ –¥–ª—è —Ä–∞–∑–±–∏–µ–Ω–∏—è
 * @param {number} [chunkSize=4] - –†–∞–∑–º–µ—Ä —á–∞—Å—Ç–µ–π (–ø–æ —É–º–æ–ª—á–∞–Ω–∏—é 4)
 * @returns {Array<string[]>} - –ú–∞—Å—Å–∏–≤ —Å —á–∞—Å—Ç—è–º–∏ —Ü–∏—Ñ—Ä
 */
function splitNumberIntoChunks(n, chunkSize = 4) {
    // 1. –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º —á–∏—Å–ª–æ –≤ —Å—Ç—Ä–æ–∫—É –∏ —Ä–∞–∑–±–∏–≤–∞–µ–º –Ω–∞ –º–∞—Å—Å–∏–≤ —Ü–∏—Ñ—Ä
    const digits = String(n).split('');
    
    // 2. –°–æ–∑–¥–∞–µ–º –º–∞—Å—Å–∏–≤ –¥–ª—è —Ä–µ–∑—É–ª—å—Ç–∞—Ç–∞
    const result = [];
    
    // 3. –†–∞–∑–±–∏–≤–∞–µ–º –Ω–∞ —á–∞—Å—Ç–∏ —É–∫–∞–∑–∞–Ω–Ω–æ–≥–æ —Ä–∞–∑–º–µ—Ä–∞
    for (let i = 0; i < digits.length; i += chunkSize) {
        const chunk = digits.slice(i, i + chunkSize);
        result.push(chunk);
    }
    
    return result;
}

// –ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è:
console.log(splitNumberIntoChunks(1234567890)); 
// [['1','2','3','4'], ['5','6','7','8'], ['9','0']]
console.log(splitNumberIntoChunks(1234567890, 3));
// [['1','2','3'], ['4','5','6'], ['7','8','9'], ['0']]
```
