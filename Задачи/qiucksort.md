/**
 * –ë—ã—Å—Ç—Ä–∞—è —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ (QuickSort) —Å –ø–æ–¥—Å—á–µ—Ç–æ–º –æ–ø–µ—Ä–∞—Ü–∏–π

 * @returns –û—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã–π –º–∞—Å—Å–∏–≤

function quickSort<T>(arr: T[]): T[] {
  // –ë–∞–∑–æ–≤—ã–π —Å–ª—É—á–∞–π: –º–∞—Å—Å–∏–≤—ã –¥–ª–∏–Ω–æ–π 0 –∏–ª–∏ 1 —É–∂–µ –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω—ã
  if (arr.length <= 1) return arr;
  
  // –í—ã–±–∏—Ä–∞–µ–º –æ–ø–æ—Ä–Ω—ã–π —ç–ª–µ–º–µ–Ω—Ç (—Å–µ—Ä–µ–¥–∏–Ω–∞ –º–∞—Å—Å–∏–≤–∞)
  const pivotIndex = Math.floor(arr.length / 2);
  const pivot = arr[pivotIndex];
  
  // –†–∞–∑–¥–µ–ª—è–µ–º —ç–ª–µ–º–µ–Ω—Ç—ã –Ω–∞ –ø–æ–¥–º–∞—Å—Å–∏–≤—ã
  const less: T[] = [];
  const greater: T[] = [];
  
  for (let i = 0; i < arr.length; i++) {
    // –ü—Ä–æ–ø—É—Å–∫–∞–µ–º —Å—Ä–∞–≤–Ω–µ–Ω–∏–µ –æ–ø–æ—Ä–Ω–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞ —Å —Å–∞–º–∏–º —Å–æ–±–æ–π
    if (i === pivotIndex) continue;
    
    // –†–∞—Å–ø—Ä–µ–¥–µ–ª—è–µ–º —ç–ª–µ–º–µ–Ω—Ç—ã –ø–æ –ø–æ–¥–º–∞—Å—Å–∏–≤–∞–º
    if (arr[i] < pivot) {
      less.push(arr[i]);
    } else {
      greater.push(arr[i]);
    }
  }
  
  // –†–µ–∫—É—Ä—Å–∏–≤–Ω–æ —Å–æ—Ä—Ç–∏—Ä—É–µ–º –ø–æ–¥–º–∞—Å—Å–∏–≤—ã –∏ –æ–±—ä–µ–¥–∏–Ω—è–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç
  return [...quickSort(less), pivot, ...quickSort(greater)];
}

// –ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è

console.log(quickSort(numbers));
