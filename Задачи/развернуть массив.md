function reverseArray(arr) {
    // 1. –°–æ–∑–¥–∞–µ–º –∫–æ–ø–∏—é –º–∞—Å—Å–∏–≤–∞ –¥–ª—è –Ω–µ–∏–∑–º–µ–Ω—è–µ–º–æ—Å—Ç–∏ –æ—Ä–∏–≥–∏–Ω–∞–ª–∞
    const reversed = [...arr];
    
    // 2. –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É–µ–º —É–∫–∞–∑–∞—Ç–µ–ª–∏

    }

}
