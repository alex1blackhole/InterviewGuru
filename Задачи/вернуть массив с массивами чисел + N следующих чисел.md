```js
/**
 * –§—É–Ω–∫—Ü–∏—è —Ä–∞–∑–±–∏–≤–∞–µ—Ç –º–∞—Å—Å–∏–≤ –Ω–∞ –ø–æ–¥–º–∞—Å—Å–∏–≤—ã –∑–∞–¥–∞–Ω–Ω–æ–≥–æ —Ä–∞–∑–º–µ—Ä–∞
 * @param {Array} arr - –ò—Å—Ö–æ–¥–Ω—ã–π –º–∞—Å—Å–∏–≤
 * @param {number} n - –†–∞–∑–º–µ—Ä —á–∞–Ω–∫–æ–≤ (—à–∞–≥)
 * @returns {Array} - –ú–∞—Å—Å–∏–≤ —Å –ø–æ–¥–º–∞—Å—Å–∏–≤–∞–º–∏
 */
function chunkArray(arr, n) {
    // 1. –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –ø—É—Å—Ç–æ–π –º–∞—Å—Å–∏–≤ –∏–ª–∏ –Ω–µ–≤–∞–ª–∏–¥–Ω—ã–π —à–∞–≥
    if (!arr.length || n <= 0) return [];
    
    // 2. –ï—Å–ª–∏ —à–∞–≥ –±–æ–ª—å—à–µ –¥–ª–∏–Ω—ã –º–∞—Å—Å–∏–≤–∞ - –≤–æ–∑–≤—Ä–∞—â–∞–µ–º –≤–µ—Å—å –º–∞—Å—Å–∏–≤
    if (n >= arr.length) return [arr];
    
    // 3. –°–æ–∑–¥–∞–µ–º —á–∞–Ω–∫–∏ —Å –ø–æ–º–æ—â—å—é reduce
    return arr.reduce((acc, _, index) => {
        // 4. –ü—Ä–æ–≤–µ—Ä—è–µ–º, –º–æ–∂–µ–º –ª–∏ –≤–∑—Ç—å n —ç–ª–µ–º–µ–Ω—Ç–æ–≤
        if (index + n <= arr.length) {
            return [...acc, arr.slice(index, index + n)];
        }
        return acc;
    }, []);
}

// –ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è:
// console.log(chunkArray([1, 2, 3, 4], 2)); // [[1, 2], [2, 3], [3, 4]]
```
